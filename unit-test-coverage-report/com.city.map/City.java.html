package com.city.map;

import java.util.HashSet;
import java.util.Objects;
import java.util.Set;
import java.util.stream.Collectors;

public class City {

	private String name;

	private Set<City> checkMapped = new HashSet<>();

	private City(String name) {
		Objects.requireNonNull(name);
		this.name = name.trim().toUpperCase();
	}

	private City() {

	}
	
	public static City getCity(String name) {
		return new City(name);
	}

	public City checkMappedCity(City city) {
		checkMapped.add(city);
		return this;
	}
	
	@Override
	public String toString() {
		return "City [name=" + name + ", checkMapped=" + checkMapped + "]";
	}

	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

	public Set<City> getCheckMapped() {
		return checkMapped;
	}

	public void setCheckMapped(Set<City> checkMapped) {
		this.checkMapped = checkMapped;
	}

	public String prettyPrint() {
		return checkMapped.stream().map(City::getName).collect(Collectors.joining(","));
	}

	@Override
	public int hashCode() {
		/*
		 * final int prime = 31; int result = 1; result = prime * result + ((name ==
		 * null) ? 0 : name.hashCode());
		 */
		return Objects.hash(name);
	}

	@Override
	public boolean equals(Object obj) {
		if (this == obj)
			return true;
		if (obj == null)
			return false;
		if (getClass() != obj.getClass())
			return false;
		City city = (City) obj;
		return Objects.equals(name, city.name);
	}

}
